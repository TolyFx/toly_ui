class TagDemo2 extends StatefulWidget {
  const TagDemo2({super.key});

  @override
  State<TagDemo2> createState() => _TagDemo2State();
}

class _TagDemo2State extends State<TagDemo2> {
  List<String> tags = ['上进', '努力', '成功', '创新'];

  @override
  Widget build(BuildContext context) {
    return Column(
      crossAxisAlignment: CrossAxisAlignment.start,
      children: [
        Wrap(
          spacing: 8,
          runSpacing: 8,
          children: [
            Tag(
              closable: true,
              onClose: () => print('标签已关闭'),
              child: const Text('可关闭'),
            ),
            Tag(
              closable: true,
              icon: const Icon(Icons.star, size: 12),
              color: Colors.orange,
              child: const Text('带图标'),
            ),
            Tag(
              disabled: true,
              closable: true,
              child: const Text('禁用状态'),
            ),
          ],
        ),
        const SizedBox(height: 16),
        const Text('动态标签列表：'),
        const SizedBox(height: 8),
        Wrap(
          spacing: 8,
          runSpacing: 8,
          children: tags.map((tag) {
            return Tag(
              closable: true,
              color: Colors.blue,
              onClose: () {
                setState(() {
                  tags.remove(tag);
                });
              },
              child: Text(tag),
            );
          }).toList(),
        ),
      ],
    );
  }
}
